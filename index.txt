    const cellVal=cell=>{if(!cell)return"";if(cell.f!=null&&String(cell.f)!=="")return String(cell.f);if(cell.v!=null)return String(cell.v);return"";};
    window[cb]=function(resp){clean();try{
      const t=resp.table;
      const data=(t.rows||[]).map(r=>(r.c||[]).map(cellVal));
      const hi=hr-1;
      const headers=(data[hi]||[]).map((h,i)=>(h&&String(h).trim())?String(h).trim():"欄"+(i+1));
      resolve({headers,rows:data.slice(hi+1)});
    }catch(e){reject(new Error("解析失敗,請確認是有效的試算表"));}};
    s.onerror=()=>{if(!done){clean();reject(new Error("無法連線(請確認試算表已開放檢視)"));}};
    s.src=src;document.head.appendChild(s);
  });
}
async function fetchSheet(url,headerRow){
  const info=extractSheet(url);
  if(info)return await fetchGvizTable(info.id,info.gid,headerRow);
  const res=await fetch(url,{method:"GET",redirect:"follow"});
  if(!res.ok)throw new Error("連線失敗 ("+res.status+")");
  const data=await res.json();
  if(!data||!Array.isArray(data.headers))throw new Error("回傳格式不符,請確認 Apps Script 程式碼");
  const hr=Math.max(1,parseInt(headerRow,10)||1);
  if(hr===1)return {headers:data.headers,rows:data.rows||[]};
  const all=[data.headers,...(data.rows||[])];
  return {headers:all[hr-1]||[],rows:all.slice(hr)};
}
function detectTitleIdx(headers){
  const idx=[];
  headers.forEach((h,i)=>{if(/商\s*[户戶]\s*名/.test(String(h)))idx.push(i);});
  if(idx.length)return idx;
  // 退而求其次:含「名稱/名」的欄,再不行就第一欄
  const alt=[];headers.forEach((h,i)=>{if(/名/.test(String(h)))alt.push(i);});
  return alt.length?alt:[0];
}
function colLetterToIdx(s){
  s=String(s).trim().toUpperCase();
  if(!/^[A-Z]+$/.test(s))return -1;
  let n=0;for(let i=0;i<s.length;i++){n=n*26+(s.charCodeAt(i)-64);}
  return n-1;
}
// 有指定主標題欄就用它(欄字母如 B,或直接打欄名);否則自動偵測商戶名
function resolveTitleIdx(headers,titleCol){
  if(titleCol&&String(titleCol).trim()){
    const idx=[];
    String(titleCol).split(/[,，]/).forEach(tok=>{
      tok=tok.trim();if(!tok)return;
      const ci=colLetterToIdx(tok);
      if(ci>=0){idx.push(ci);return;}
      headers.forEach((h,i)=>{if(String(h).includes(tok))idx.push(i);});
    });
    const uniq=[...new Set(idx)].filter(i=>i>=0&&i<headers.length);
    if(uniq.length)return uniq;
  }
  return detectTitleIdx(headers);
}
async function syncProject(p){
  if(syncing||!isSynced(p))return;
  syncing=true;p.syncErr=null;
  if(view.name==="project"&&view.pid===p.id)renderProject();
  try{

    const titleIdx=resolveTitleIdx(data.headers,p.sheet.titleCol);
    p.records=mapRows(data.headers,data.rows||[],titleIdx,-1);
    p.lastSync=Date.now();save();
  }catch(err){p.syncErr=String(err.message||err);}
  syncing=false;render();
}

function fmtAgo(ts){if(!ts)return"";const d=Math.floor((Date.now()-ts)/60000);if(d<1)return"剛剛";if(d<60)return d+" 分鐘前";const h=Math.floor(d/60);if(h<24)return h+" 小時前";return Math.floor(h/24)+" 天前";}
function fmtWhen(p){
  if(isSynced(p))return p.lastSync?("同步 "+fmtAgo(p.lastSync)):"待同步";
  if(!p.records.length)return"尚無記錄";
  const last=Math.max(...p.records.map(r=>r.createdAt||0));
  const d=Math.floor((Date.now()-last)/86400000);
  return d<=0?"今天更新":d===1?"昨天更新":d<30?d+" 天前更新":"較久未更新";
}

    if(!title&&!note){modalRoot.querySelector("#r-title").focus();return;}
    if(r){r.title=title;r.date=date;r.note=note;}else p.records.push({id:uid(),title,date,note,createdAt:Date.now()});
    save();closeModal();renderProject();
  };
}

}
